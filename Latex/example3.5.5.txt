\subsection*{Example 3.5.5: Volume with Curved Bounds}

Find the volume of the solid region
\[
E = \{(x, y, z) \mid 0 \leq x \leq 1,\ x^2 \leq y \leq x,\ 0 \leq z \leq xy \}.
\]

\begin{figure}[h]
    \centering
    \includegraphics[width=0.65\textwidth]{curved_bounds_volume.png}
    \caption{Solid region with curved bounds: $0 \leq x \leq 1$, $x^2 \leq y \leq x$, $0 \leq z \leq xy$.}
    \label{fig:curved_bounds}
\end{figure}

The volume is given by the triple integral
\[
V = \iiint_E 1 \, dV = \int_{0}^{1} \int_{x^2}^{x} \int_{0}^{xy} \, dz \, dy \, dx.
\]

\subsubsection*{Step 1: Integrate with respect to $z$}

\[
\int_{0}^{xy} 1 \, dz = \left[ z \right]_{0}^{xy} = xy.
\]

The integral simplifies to
\[
V = \int_{0}^{1} \int_{x^2}^{x} xy \, dy \, dx.
\]

\subsubsection*{Step 2: Integrate with respect to $y$}

\[
\int_{x^2}^{x} xy \, dy = x \int_{x^2}^{x} y \, dy = x \left[ \frac{y^2}{2} \right]_{x^2}^{x} = x \left( \frac{x^2}{2} - \frac{(x^2)^2}{2} \right) = x \left( \frac{x^2}{2} - \frac{x^4}{2} \right) = \frac{x^3}{2} - \frac{x^5}{2}.
\]

The integral now becomes
\[
V = \int_{0}^{1} \left( \frac{x^3}{2} - \frac{x^5}{2} \right) \, dx.
\]

\subsubsection*{Step 3: Integrate with respect to $x$}

\[
V = \frac{1}{2} \int_{0}^{1} \left( x^3 - x^5 \right) \, dx = \frac{1}{2} \left[ \frac{x^4}{4} - \frac{x^6}{6} \right]_{0}^{1} = \frac{1}{2} \left( \frac{1}{4} - \frac{1}{6} \right) = \frac{1}{2} \left( \frac{3}{12} - \frac{2}{12} \right) = \frac{1}{2} \cdot \frac{1}{12} = \frac{1}{24}.
\]
